var hanoi = function (dist,src,aux,dst){
   if(dist > 0){
    hanoi(dist-1,src,dst,aux);
    console.log('move dist ' + Date.now(),dist + ' from ' + src + ' to ' + dst);
    hanoi(dist-1,aux,src,dst);
   }
};
hanoi(2,'From','Aux','To');
